def load_sync_timestamps() -> Dict[str, datetime]:
    """Loads the store-to-sync-timestamp mapping from a file."""
    if SYNC_TIMESTAMPS_FILE.exists():
        try:
            with open(SYNC_TIMESTAMPS_FILE, "rb") as f:
                return pickle.load(f)
        except (pickle.UnpicklingError, EOFError, FileNotFoundError):
            return {}
    return {}


def save_sync_timestamp(store_key: str):
    """Updates the sync timestamp for a given store and saves it to a file."""
    timestamps = load_sync_timestamps()
    timestamps[store_key] = datetime.now(timezone.utc)
    try:
        with open(SYNC_TIMESTAMPS_FILE, "wb") as f:
            pickle.dump(timestamps, f)
    except IOError as e:
        logger.error(f"Failed to save sync timestamp for {store_key}: {e}")

def safe_get(data: Any, path: str, default=None) -> Any:
    if not isinstance(data, dict):
        return default
    keys = path.split(".")
    val = data
    for k in keys:
        if not isinstance(val, dict):
            return default
        val = val.get(k, default)
        if val is None:
            return default
    return val


def safe_parse_date_iso(s: Optional[str]) -> Optional[datetime]:
    if not s or not isinstance(s, str):
        return None
    try:
        return datetime.fromisoformat(s.replace("Z", "+00:00"))
    except Exception:
        return None


def format_date(dt: Optional[datetime], fmt: str = "%Y-%m-%d %H:%M") -> str:
    if not dt:
        return "-"
    return dt.strftime(fmt)


def compute_days_since(from_date: Optional[datetime]) -> int:
    if not from_date:
        return 0
    now_utc = datetime.now(timezone.utc)
    d = from_date.astimezone(timezone.utc) if from_date.tzinfo else from_date.replace(tzinfo=timezone.utc)
    return (now_utc - d).days


def get_event_date(rma_data: dict, event_name: str) -> Optional[datetime]:
    events = rma_data.get("events", []) or safe_get(rma_data, "rmaSummary.events", [])
    for ev in (events or []):
        if ev.get("eventName") == event_name:
            return safe_parse_date_iso(ev.get("eventDate"))
    return None


def get_received_date(rma_data: dict) -> Optional[datetime]:
    ev = get_event_date(rma_data, "SHIPMENT_RECEIVED")
    if ev:
        return ev
    for c in (safe_get(rma_data, "comments", []) or []):
        if "RECEIVED" in str(c.get("htmlText", "")).upper():
            d = safe_parse_date_iso(c.get("datetime"))
            if d:
                return d
    return None


def get_credit_date(rma_data: dict) -> Optional[datetime]:
    for t in (safe_get(rma_data, "transactions", []) or []):
        tt = str(t.get("type", "")).lower()
        ts = str(t.get("status", "")).lower()
        if any(x in tt for x in ("refund", "credit", "reimbursement")) and ts in ("success", "completed", "succeeded"):
            d = safe_parse_date_iso(t.get("createdAt"))
            if d:
                return d
    return None

def get_last_rma_update(rma_data: dict) -> str:
    comments = safe_get(rma_data, "comments", []) or []
    if not comments:
        return ""
    latest = max(comments, key=lambda c: safe_parse_date_iso(c.get("datetime")) or datetime.min)
    dt = safe_parse_date_iso(latest.get("datetime"))
    raw = re.sub(r"<[^>]+>", "", latest.get("htmlText", ""))
    plain = html_lib.unescape(raw).strip()[:100]
    prefix = dt.strftime("%d-%m-%y %H:%M") if dt else ""
    return f"{prefix} - {plain}" if prefix else plain


def is_resolution_actioned(json_data: dict) -> str:
    for t in (safe_get(json_data, "transactions", []) or []):
        if str(t.get("status", "")).lower() in ("success", "completed", "succeeded"):

    for eo in (safe_get(json_data, "exchangeOrders", []) or []):
        if eo.get("orderName") and not str(eo.get("orderName")).startswith("#"):
            return "Yes"
    return "No"

def hex_to_rgb_str(hex_color: str) -> str:
    """Converts a hex color string like #RRGGBB to an 'R,G,B' string."""
    if not hex_color or not hex_color.startswith("#"):
        return "0,0,0"
    hex_color = hex_color.lstrip("#")
    if len(hex_color) != 6:
        return "0,0,0"
    try:
        return ",".join(str(int(hex_color[i:i + 2], 16)) for i in (0, 2, 4))
    except ValueError:
        return "0,0,0"

# ③ RATE LIMITER
# ─────────────────────────────────────────────
class RateLimiter:
    def __init__(self, calls_per_second: float):
        self.interval = 1.0 / calls_per_second if calls_per_second > 0 else 0
        self.last_call = 0.0
        self.lock = threading.Lock()

    def wait_if_needed(self):
        with self.lock:
            elapsed = time.time() - self.last_call
            sleep_time = self.interval - elapsed
            if sleep_time > 0:
                time.sleep(sleep_time)
            self.last_call = time.time()


returngo_limiter = RateLimiter(RG_RPS)

# ─────────────────────────────────────────────
# ④ RATE LIMIT TRACKING
# ─────────────────────────────────────────────
def update_rate_limit_info(headers: Mapping[str, str]):
    with RATE_LIMIT_LOCK:
        if "X-RateLimit-Remaining" in headers:
            RATE_LIMIT_INFO["remaining"] = headers["X-RateLimit-Remaining"]
        if "X-RateLimit-Limit" in headers:
            RATE_LIMIT_INFO["limit"] = headers["X-RateLimit-Limit"]
        if "X-RateLimit-Reset" in headers:
            try:
                RATE_LIMIT_INFO["reset"] = datetime.fromtimestamp(int(headers["X-RateLimit-Reset"]), tz=timezone.utc)
            except Exception:
                pass
        RATE_LIMIT_INFO["updated_at"] = datetime.now(timezone.utc)

def get_requests_session() -> requests.Session:
    s = requests.Session()
    retry = Retry(total=3, status_forcelist=[429, 500, 502, 503, 504],
                  allowed_methods=["GET", "POST", "PUT"], backoff_factor=2)
    s.mount("https://", HTTPAdapter(max_retries=retry))
    return s


def get_pn_session() -> requests.Session:
    s = requests.Session()
    retry = Retry(total=3, status_forcelist=[429, 500, 502, 503, 504],
                  allowed_methods=["GET"], backoff_factor=1, respect_retry_after_header=True)
    s.mount("https://", HTTPAdapter(max_retries=retry))
    return s

def rg_headers(store_key: str) -> dict:
    return {
        "x-api-key":    RETURNGO_API_KEY,
        "x-shop-name":  STORE_CONFIG[store_key]["shopify_url"],
        "Content-Type": "application/json",
    }


def fetch_returngo_rmas(store_key: str, status: Optional[str] = None, since_dt: Optional[datetime] = None, max_pages: int = 9999) -> List[dict]:
    all_rmas: List[dict] = []
    session = get_requests_session()
    page, cursor = 1, None

    while page <= max_pages:
        returngo_limiter.wait_if_needed()
        params: dict = {"pagesize": "100"}
        if cursor:
            params["cursor"] = cursor
        if status:
            params["status"] = status
        if since_dt:
            # Overlap to prevent missing records at the boundary
            since_dt_overlap = since_dt - timedelta(minutes=15)
            params["rma_updated_at"] = f"gte:{since_dt_overlap.isoformat()}"

        try:
            t0 = time.time()
            resp = session.get(api_url("/rmas"), headers=rg_headers(store_key), params=params, timeout=60)
            _update_perf(time.time() - t0)
            update_rate_limit_info(resp.headers)

            if resp.status_code == 429:
                logger.warning(f"Rate limit hit page {page} for {store_key}. Sleeping 60s.")
                RATE_LIMIT_HIT.set()
                time.sleep(60)
                RATE_LIMIT_HIT.clear()
                continue

            resp.raise_for_status()
            data = resp.json()
            rmas = data.get("rmas", [])
            if not rmas:
                break
            all_rmas.extend(rmas)
            cursor = data.get("next_cursor")
            if not cursor:
                break
            page += 1

        except requests.exceptions.RequestException as e:
            logger.error(f"[ReturnGO] Network error for {store_key} on page {page}: {e}")
            break
        except json.JSONDecodeError as e:
            logger.error(f"[ReturnGO] Failed to decode JSON for {store_key} on page {page}: {e}")
            break
        except Exception as e:
            logger.error(f"[ReturnGO] Unexpected error for {store_key} on page {page}: {e}")
            break

    logger.info(f"[ReturnGO] {store_key}: fetched {len(all_rmas)} RMAs.")
    return all_rmas


def get_rma_by_id(store_key: str, rma_id: str) -> Optional[dict]:

        logger.error(f"[ReturnGO] Unexpected error getting RMA {rma_id} for {store_key}: {e}")

def post_rma_comment(store_key: str, rma_id: str, comment_text: str) -> bool:
    returngo_limiter.wait_if_needed()
    session = get_requests_session()
    try:
        t0 = time.time()
        resp = session.post(
            api_url(f"/rma/{rma_id}/comment"),
            headers=rg_headers(store_key),
            json={"comment": comment_text},
            timeout=60,
        )
        _update_perf(time.time() - t0)
        update_rate_limit_info(resp.headers)
        resp.raise_for_status()
        return True
    except requests.exceptions.RequestException as e:
        logger.error(f"[ReturnGO] Network error posting comment to RMA {rma_id}: {e}")
        return False
    except Exception as e:
        logger.error(f"[ReturnGO] Unexpected error posting comment to RMA {rma_id}: {e}")
        return False


def push_tracking_update(store_key: str, rma_id: str, shipment_id: str, tracking_number: str) -> Tuple[bool, str]:
    payload = {
        "status":       "LabelCreated",
        "carrierName":  "CourierGuy",
        "trackingNumber": tracking_number,
        "trackingURL":  f"https://optimise.parcelninja.com/shipment/track?WaybillNo={tracking_number}",
    }
    try:
        resp = requests.put(
            api_url(f"/shipment/{shipment_id}"),
            headers=rg_headers(store_key),
            json=payload,
            timeout=15,
        )
        if resp.status_code in (200, 201):
            return True, "Success"
        return False, f"API Error {resp.status_code}: {resp.text}"
    except Exception as e:
        return False, str(e)

def _pn_headers(store_key: str) -> dict:

        "Accept":          "application/json",
        "Authorization":   f"Basic {_pn_auth_header(store_key)}",
        "X-Store-Id":      STORE_CONFIG[store_key]["pn_store_id"],
    }


@st.cache_data(ttl=3600, show_spinner=False)
def fetch_parcelninja_inbounds(store_key: str, start_date: datetime, end_date: datetime) -> List[dict]:
    """Fetch inbound list from Parcelninja for cross-referencing with RMAs."""
    headers = _pn_headers(store_key)
    params = {
        "pageSize": 500,
        "page":     1,
        "startDate": start_date.strftime("%Y%m%d"),
        "endDate":   end_date.strftime("%Y%m%d"),
        "col":       4,
        "colOrder":  "desc",
    }
    all_inbounds: List[dict] = []
    try:
        while True:
            resp = requests.get(f"{PN_BASE}/inbounds", headers=headers, params=params, timeout=20)
            if resp.status_code != 200:
                logger.warning(f"[PN Inbounds] {store_key}: HTTP {resp.status_code}")
                break
            data = resp.json()
            batch = data.get("inbounds", [])
            if not batch:
                break
            all_inbounds.extend(batch)
            if not data.get("next"):
                break
            params["page"] += 1
    except Exception as e:
        logger.error(f"[PN Inbounds] {store_key}: {e}")
    return all_inbounds

    except Exception as e:
        logger.warning(f"[PN Events] {store_key}/{inbound_id}: {e}")
        return None


def _parse_pn_ts(ts: Optional[str]) -> Optional[datetime]:
    if not ts:
        return None
    try:
        s = str(ts)
        if len(s) == 14:
            return datetime.strptime(s, "%Y%m%d%H%M%S").replace(tzinfo=timezone.utc)
        if len(s) == 8:
            return datetime.strptime(s, "%Y%m%d").replace(tzinfo=timezone.utc)
        return datetime.fromisoformat(s.replace("Z", "+00:00"))
    except Exception:
        return None


def _fmt_pn_ts(ts: Optional[str]) -> str:
    dt = _parse_pn_ts(ts)
    return dt.strftime("%d %b %Y %H:%M") if dt else "-"


def fetch_pn_inbounds_enriched(store_key: str, days: int = 14) -> List[dict]:
    """Fetch + event-enrich all inbounds for a store. Used by Inbound Control Centre tab."""
    end_dt   = datetime.now()
    start_dt = end_dt - timedelta(days=days)
    start_s  = start_dt.strftime("%Y%m%d")  # Rule 2
    end_s    = end_dt.strftime("%Y%m%d")    # Rule 2

    headers = _pn_headers(store_key)
    params = {
        "pageSize": 100,
        "page":     1,
        "startDate": start_s,
        "endDate":   end_s,
        "col":       4,
        "colOrder":  "desc",
    }
    summaries: List[dict] = []
    try:
        resp = get_pn_session().get(f"{PN_BASE}/inbounds", headers=headers, params=params, timeout=20)
        if resp.status_code == 200:
            summaries = resp.json().get("inbounds", [])
    except Exception as e:
        logger.error(f"[PN Enriched List] {store_key}: {e}")
        return []

    cfg = STORE_CONFIG[store_key]
    records: List[dict] = []
    session = get_pn_session()

    def enrich(summary: dict) -> Optional[dict]:
        ib_id = summary.get("id")
        detail = _fetch_pn_inbound_events(store_key, ib_id, session) or summary
        events = detail.get("events", [])
        # Rule 1: derive status from events, newest first
        latest_ev    = events[0] if events else {}
        status_code  = latest_ev.get("code", 0)
        status_desc  = latest_ev.get("description", "Unknown")
        last_ev_ts   = _parse_pn_ts(latest_ev.get("timeStamp"))
        status_meta  = PN_INBOUND_STATUS.get(status_code, {"label": status_desc, "emoji": "❓", "color": "#6b7280"})

        items          = detail.get("items", []) or []
        expected_units = sum(int(i.get("qty") or 0) for i in items)
        d_info         = detail.get("deliveryInfo", {}) or {}
        p_info         = detail.get("pickupInfo", {}) or {}

        def _est_arrival():
            raw = d_info.get("estimatedArrivalDate")
            if raw:
                dt = _parse_pn_ts(str(int(raw)) if isinstance(raw, float) else str(raw))
                if dt:
                    return dt.strftime("%d %b %Y")
            return "—"

        return {
            "store_key":      store_key,
            "brand":          cfg["display_name"],
            "pn_id":          detail.get("id", ib_id),
            "client_ref":     detail.get("clientId", "—") or "—",
            "supplier_ref":   detail.get("supplierReference", "—") or "—",
            "inbound_type":   (detail.get("type") or {}).get("description", "—"),
            "status_code":    status_code,
            "status_label":   status_meta["label"],
            "status_emoji":   status_meta["emoji"],
            "status_color":   status_meta["color"],
            "is_complete":    status_code in (203, 204),
            "is_variance":    status_code == 204,
            "is_in_progress": status_code not in (203, 204, 0),
            "expected_units": expected_units,
            "item_count":     len(items),
            "est_arrival":    _est_arrival(),
            "waybill":        d_info.get("trackingNo", "—") or "—",
            "courier":        d_info.get("courierName", "—") or "—",
            "last_event_ts":  last_ev_ts,
            "last_event_str": last_ev_ts.strftime("%d %b %Y %H:%M") if last_ev_ts else "—",
            "create_date":    _fmt_pn_ts(str(detail.get("createDate", ""))),
            "raw_items":      items,
            "raw_events":     events,
        }

    with concurrent.futures.ThreadPoolExecutor(max_workers=4) as ex:
        for rec in ex.map(enrich, summaries):
            if rec:
                records.append(rec)

    records.sort(
        key=lambda r: r["last_event_ts"] or datetime.min.replace(tzinfo=timezone.utc),
        reverse=True,
    )
    return records


@st.cache_data(ttl=1800, show_spinner=False)
def fetch_pn_latest_status_by_client(store_key: str, start_date: datetime, end_date: datetime) -> Dict[str, Dict[str, str]]:
    """Return latest inbound event status per Parcelninja clientId for a date window."""
    inbounds = fetch_parcelninja_inbounds(store_key, start_date, end_date)
    if not inbounds:
        return {}

    latest_inbound_by_client: Dict[str, dict] = {}
    for ib in inbounds:
        client_id = str(ib.get("clientId", "")).strip()
        if not client_id:
            continue
        create_dt = _parse_pn_ts(str(ib.get("createDate", "")))
        current = latest_inbound_by_client.get(client_id)
        current_dt = _parse_pn_ts(str(current.get("createDate", ""))) if current else None
        if create_dt and (current_dt is None or create_dt > current_dt):
            latest_inbound_by_client[client_id] = ib

    client_status_map: Dict[str, Dict[str, str]] = {}
    for client_id, summary in latest_inbound_by_client.items():
        inbound_id = summary.get("id")
        if not inbound_id:
            continue
        detail = _fetch_pn_inbound_events(store_key, inbound_id)
        events = (detail or {}).get("events", [])
        latest_ev = events[0] if events else {}
        client_status_map[client_id] = {
            "inbound_status": latest_ev.get("description") or "-",
            "inbound_update_date": _fmt_pn_ts(latest_ev.get("timeStamp")) if latest_ev else "-",
        }

    return client_status_map


# ─────────────────────────────────────────────
# ⑧ PARCELNINJA TRACKING SCRAPER (mirrors Levi's)
# ─────────────────────────────────────────────
def scrape_parcel_ninja_status(tracking_number: str) -> str:
    if not tracking_number:
        return "No tracking number"
    clean = tracking_number.replace(" ↻", "").strip()
    url = f"https://optimise.parcelninja.com/shipment/track?WaybillNo={clean}"
    try:
        resp = requests.get(
            url,
            headers={
                "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
                "Accept": "text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8",
            },
            timeout=10,
        )
        if resp.status_code == 404:
            return "Tracking not found (404)"
        if resp.status_code in (401, 403):
            return f"Scraping blocked ({resp.status_code})"
        if resp.status_code == 429:
            return "Scraping rate limited"
        resp.raise_for_status()
        content = resp.text

        patterns = [
            re.compile(r"((?:Mon|Tue|Wed|Thu|Fri|Sat|Sun),\s+\d{1,2}\s+[A-Za-z]{3}\s+\d{2}:\d{2})\s+(?:<[^>]+>)*\s*([^<]+)", re.MULTILINE),
            re.compile(r"((?:Mon|Tue|Wed|Thu|Fri|Sat|Sun),\s+\d{1,2}\s+[A-Za-z]{3}\s+\d{2}:\d{2})\s+(.+?)(?=\n|$)", re.MULTILINE),
        ]
        for pattern in patterns:
            matches = [(m[0].strip(), re.sub(r"\s+", " ", html_lib.unescape(re.sub(r"<[^>]+>", "", m[1]))).strip())
                       for m in pattern.findall(content) if len(m) == 2 and m[0] and m[1]]
            if matches:
                return f"{matches[0][0]} | {matches[0][1]}"

        for kw in ["Courier Cancelled", "Delivered", "In Transit", "Out for Delivery", "Submitted to Courier"]:
            if re.search(kw, content, re.IGNORECASE):
                return kw

        return "Status not found on page"

    except requests.exceptions.RequestException as e:
        return f"Scraping request failed: {type(e).__name__}"
    except Exception as e:
        logger.error(f"[Scraper] {clean}: {e}")
        return "Scraping check failed"


def get_shipment_status(rma_data: dict) -> Optional[str]:
    shipments = safe_get(rma_data, "shipments", [])
    if not shipments or not isinstance(shipments, list):
        return None
    tn = safe_get(shipments[0], "trackingNumber") if shipments else None
    if not tn:
        return None
    status = scrape_parcel_ninja_status(tn)
    return status if status and status != "N/A" else "N/A"

def _load_file(path: Path, cls):
    if path.exists():
        try:
            with open(path, "rb") as f:
                return pickle.load(f)
        except Exception:
            pass
    return cls()


def _save_file(path: Path, obj):
    try:
        with open(path, "wb") as f:
            pickle.dump(obj, f)
    except Exception as e:
        logger.error(f"Save failed: {e}")


def _update_perf(api_time: float):
    metrics = _load_file(PERFORMANCE_FILE, PerformanceMetrics)
    metrics.api_call_times.append(api_time)
    metrics.api_call_times = metrics.api_call_times[-100:]
    if metrics.api_call_times:
        metrics.avg_response_time = sum(metrics.api_call_times) / len(metrics.api_call_times)
    _save_file(PERFORMANCE_FILE, metrics)

def _process_rma_for_df(rma_summary: dict, store_key: str) -> Optional[dict]:

        "created_at":    safe_parse_date_iso(detail.get("createdAt")),
        "json_data":     detail,
        "courier_status": courier_status,
    }


@st.cache_data(ttl=CACHE_EXPIRY_SECS, show_spinner=False)
def fetch_and_cache_store_data(store_key: str) -> pd.DataFrame:
    """
    Performs a FULL data fetch for a store. Fetches all active RMAs, enriches them,
    and returns a DataFrame. This function is cached for 12 hours.
    """
    sync_start = time.time()
    all_rmas: List[dict] = []
    progress = st.progress(0, text=f"Syncing {STORE_CONFIG[store_key]['display_name']}…")

    for i, status in enumerate(ACTIVE_RG_STATUSES):
        progress.progress((i + 1) / (len(ACTIVE_RG_STATUSES) + 1), text=f"Fetching all {status} RMAs…")
        batch = fetch_returngo_rmas(store_key, status=status)
        all_rmas.extend(batch)

    progress.progress(0.5, text=f"Processing {len(all_rmas)} RMA details…")
    results: List[dict] = []
    with concurrent.futures.ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:
        futures = {ex.submit(_process_rma_for_df, r, store_key): r for r in all_rmas}
        for idx, fut in enumerate(concurrent.futures.as_completed(futures)):
            result = fut.result()
            if result:
                results.append(result)
            progress.progress(0.5 + (idx + 1) / max(len(all_rmas), 1) * 0.5,
                               text=f"Processing… ({idx+1}/{len(all_rmas)})")

    progress.progress(1.0, text="Sync complete!")

    if not results:
        return pd.DataFrame()

    df = pd.DataFrame(results)
    metrics = _load_file(PERFORMANCE_FILE, PerformanceMetrics)
    metrics.last_sync_duration = time.time() - sync_start
    _save_file(PERFORMANCE_FILE, metrics)
    save_sync_timestamp(store_key)
    time.sleep(1)
    progress.empty()
    return df


def update_single_rma(rma_id: str, store_key: str) -> bool:
    """Fetches a single RMA's details and updates the in-memory dataframe."""
    if not rma_id or not store_key:
        return False

    append_ops_log(f"Fetching latest data for RMA {rma_id} from {STORE_CONFIG[store_key]['display_name']}...")

    # Mock a summary object to reuse the existing processing function
    rma_summary_mock = {"rmaId": rma_id}
    updated_data = _process_rma_for_df(rma_summary_mock, store_key)

    if not updated_data:
        append_ops_log(f"Failed to fetch details for RMA {rma_id}.")
        st.error(f"Failed to fetch details for RMA {rma_id}.")
        return False

    df_key = f"rma_df_{store_key}"
    existing_df = st.session_state.get(df_key)

    if existing_df is None:
        append_ops_log(f"Cannot update RMA {rma_id}: main dataframe for {store_key} not loaded.")
        st.warning(f"Data for {STORE_CONFIG[store_key]['display_name']} is not loaded. Cannot perform update.")
        return False
    
    update_df = pd.DataFrame([updated_data])

    # Find and replace the old row
    df_without_rma = existing_df[existing_df["rma_id"] != rma_id]
    merged_df = pd.concat([df_without_rma, update_df], ignore_index=True)

    st.session_state[df_key] = merged_df
    st.session_state[f"rma_df_enriched_{store_key}"] = None # Invalidate enriched cache
    
    append_ops_log(f"RMA {rma_id} updated successfully.")
    return True


def perform_incremental_sync(store_key: str):
    """Fetches only updated RMAs and merges them into the existing dataframe."""
    df_key = f"rma_df_{store_key}"

    # 1. Check for existing data and last sync time
    existing_df = st.session_state.get(df_key)
    last_sync = load_sync_timestamps().get(store_key)

    if existing_df is None or last_sync is None:
        append_ops_log("Cannot perform incremental sync. Please do a Full Refresh first.")
        st.warning("No base data to update. Please perform a 'Full Refresh' first.")
        return

    sync_start = time.time()
    append_ops_log(f"Starting incremental sync for {STORE_CONFIG[store_key]['display_name']} since {last_sync.strftime('%H:%M:%S')}...")
    progress = st.progress(0, text=f"Fetching updates for {STORE_CONFIG[store_key]['display_name']}...")

    # 2. Fetch updated RMA summaries
    updated_rmas_list = []
    for i, status in enumerate(ACTIVE_RG_STATUSES):
        progress.progress((i + 1) / (len(ACTIVE_RG_STATUSES) + 1), text=f"Checking for {status} updates…")
        batch = fetch_returngo_rmas(store_key, status=status, since_dt=last_sync)
        updated_rmas_list.extend(batch)

    if not updated_rmas_list:
        progress.progress(1.0, text="No updates found.")
        append_ops_log("No new updates found.")
        save_sync_timestamp(store_key) # Still update timestamp to show we checked
        time.sleep(1)
        progress.empty()
        return

    # 3. Process updated RMAs to get full details
    progress.progress(0.5, text=f"Processing {len(updated_rmas_list)} updated RMA details…")
    updated_results: List[dict] = []
    with concurrent.futures.ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:
        futures = {ex.submit(_process_rma_for_df, r, store_key): r for r in updated_rmas_list}
        for idx, fut in enumerate(concurrent.futures.as_completed(futures)):
            result = fut.result()
            if result:
                updated_results.append(result)
            progress.progress(0.5 + (idx + 1) / max(len(updated_rmas_list), 1) * 0.5,
                               text=f"Processing updates… ({idx+1}/{len(updated_rmas_list)})")

    if not updated_results:
        progress.progress(1.0, text="Processing complete. No changes.")
        append_ops_log("Finished incremental sync. No data changed after detail fetch.")
        save_sync_timestamp(store_key)
        return

    # 4. Merge with existing dataframe
    update_df = pd.DataFrame(updated_results)
    updated_rma_ids = update_df["rma_id"].tolist()
    df_without_updates = existing_df[~existing_df["rma_id"].isin(updated_rma_ids)]
    merged_df = pd.concat([df_without_updates, update_df], ignore_index=True)

    # 5. Update session state
    st.session_state[df_key] = merged_df
    st.session_state[f"rma_df_enriched_{store_key}"] = None # Invalidate enriched version
    save_sync_timestamp(store_key)

    duration = time.time() - sync_start
    metrics = _load_file(PERFORMANCE_FILE, PerformanceMetrics)
    metrics.last_sync_duration = duration
    _save_file(PERFORMANCE_FILE, metrics)

    progress.progress(1.0, text="Sync complete!")
    append_ops_log(f"Incremental sync complete. Merged {len(update_df)} updates. Duration: {duration:.2f}s")
    time.sleep(1)
    progress.empty()

def force_courier_refresh(df: pd.DataFrame) -> pd.DataFrame:
    if df.empty:
        return df
    df2 = df.copy()
    mask = (
        df2["status"].str.lower().isin(["approved", "received"]) &
        df2["tracking_number"].notna() &
        (df2["tracking_number"] != "") &
        (df2["tracking_number"] != "-")
    )
    to_scrape = df2[mask]["tracking_number"].unique().tolist()
    if not to_scrape:
        return df2
    bar = st.progress(0, text=f"Refreshing {len(to_scrape)} tracking statuses…")
    new_statuses: Dict[str, str] = {}
    with concurrent.futures.ThreadPoolExecutor(max_workers=MAX_WORKERS) as ex:
        futures = {ex.submit(scrape_parcel_ninja_status, tn): tn for tn in to_scrape}
        for idx, fut in enumerate(concurrent.futures.as_completed(futures)):
            tn = futures[fut]
            new_statuses[tn] = fut.result()
            bar.progress((idx + 1) / len(to_scrape))
    df2["courier_status"] = df2["tracking_number"].map(new_statuses).fillna(df2["courier_status"])
    bar.progress(1.0, text="Refresh complete!")
    return df2

# ─────────────────────────────────────────────
def enrich_rma_dataframe(df: pd.DataFrame, store_key: str) -> pd.DataFrame:
    if df.empty:
        return df
    df = df.copy()

    df["order_name"]       = df["json_data"].apply(extract_order_name)
    df["customer_name"]    = df["json_data"].apply(extract_customer_name)
    df["tracking_number"]  = df["json_data"].apply(get_tracking_number_with_indicator)
    df["tracking_was_updated"] = df["json_data"].apply(has_tracking_update_comment)
    df["requested_date"]   = df["json_data"].apply(lambda x: get_event_date(x, "RMA_CREATED"))
    df["approved_date"]    = df["json_data"].apply(lambda x: get_event_date(x, "RMA_APPROVED"))
    df["received_date"]    = df["json_data"].apply(get_received_date)
    df["credit_date"]      = df["json_data"].apply(get_credit_date)
    df["days_since_credit"] = df["credit_date"].apply(lambda x: compute_days_since(x) if x else None)
    df["last_rma_update"]  = df["json_data"].apply(get_last_rma_update)
    df["resolution_type"]  = df["json_data"].apply(lambda x: ", ".join(get_resolution_types(x)) or "-")
    df["resolution_actioned"] = df["json_data"].apply(is_resolution_actioned)
    df["days_since_requested"] = df["requested_date"].apply(compute_days_since)
    df["is_flagged"]       = df["json_data"].apply(check_for_flagged_comment)

    # Parcelninja inbound cross-reference → client_id
    min_approved = df["approved_date"].min()
    max_approved = df["approved_date"].max()
    if pd.notna(min_approved) and pd.notna(max_approved):
        inbounds = fetch_parcelninja_inbounds(store_key, min_approved, max_approved)
        lookup: Dict[tuple, str] = {}
        for ib in inbounds:
            try:
                cds = str(ib.get("createDate", ""))
                if len(cds) >= 8:
                    cd = datetime.strptime(cds[:8], "%Y%m%d").date()
                    recipient = (ib.get("pickupInfo") or {}).get("recipient") or ""
                    client_id = ib.get("clientId", "")
                    key = (cd, recipient.lower().strip())
                    if key not in lookup:
                        lookup[key] = client_id
            except Exception:
                continue

        def find_client_id(row):
            if pd.notna(row["approved_date"]) and pd.notna(row["customer_name"]):
                cd = row["approved_date"].date()
                norm = str(row["customer_name"]).lower().strip().split(" ")[0]
                return lookup.get((cd, norm), "-")
            return "-"

        df["client_id"] = df.apply(find_client_id, axis=1)
    else:
        df["client_id"] = "-"

    df["inbound_status"] = "-"
    df["inbound_update_date"] = "-"
    try:
        non_blank_client = df["client_id"].astype(str).str.strip().replace("", "-")
        has_client = non_blank_client.ne("-")
        if has_client.any() and pd.notna(min_approved) and pd.notna(max_approved):
            status_map = fetch_pn_latest_status_by_client(
                store_key,
                min_approved - timedelta(days=7),
                max_approved + timedelta(days=7),
            )
            if status_map:
                df.loc[has_client, "inbound_status"] = non_blank_client[has_client].map(
                    lambda c: (status_map.get(c) or {}).get("inbound_status", "-")
                )
                df.loc[has_client, "inbound_update_date"] = non_blank_client[has_client].map(
                    lambda c: (status_map.get(c) or {}).get("inbound_update_date", "-")
                )
    except Exception as e:
        logger.warning(f"[PN Inbound Status Map] {store_key}: {e}")

    # Tracking status from scrape
    def fmt_tracking_status(row):
        tn  = str(row.get("tracking_number", "")).strip()
        cs  = row.get("courier_status")
        cs_str = str(cs).strip() if cs is not None else ""
        if not tn or tn == "-":
            return "No tracking number"
        if cs_str and cs_str not in ("None", ""):
            return cs_str
        return "Status not scraped"

    df["tracking_status"] = df.apply(fmt_tracking_status, axis=1)
    df["is_cc"] = df["tracking_status"].str.lower().str.contains("cancelled", na=False)

    def detect_failures(row):
        failures = []
        if "cancelled" in str(row.get("tracking_status", "")).lower():
            failures.append("COURIER_CANCELLED")
        if row.get("status", "").lower() in ("approved", "received") and not str(row.get("tracking_number", "")).strip():
            failures.append("COURIER_CANCELLED")
        if row.get("status", "").lower() == "received" and row.get("resolution_actioned", "").lower() != "yes":
            failures.append("NO_RESOLUTION")
        return ", ".join(failures)

    df["failures"] = df.apply(detect_failures, axis=1)
    return df


def get_enriched_store_df(store_key: str) -> pd.DataFrame:
    raw_key = f"rma_df_{store_key}"
    enr_key = f"rma_df_enriched_{store_key}"
    raw_df = st.session_state.get(raw_key)
    if raw_df is None:
        return pd.DataFrame()
    if st.session_state.get(enr_key) is None:
        st.session_state[enr_key] = enrich_rma_dataframe(raw_df, store_key) if not raw_df.empty else raw_df
    return st.session_state[enr_key]


def compute_counts(df: pd.DataFrame) -> Dict[str, int]:
    if df.empty:
        return {}
    c: Dict[str, int] = {}
    sl = df["status"].str.lower()
    c["Total Open"]            = len(df[sl.isin(["pending", "approved", "received"])])
    c["Pending"]               = len(df[sl == "pending"])
    c["Approved"]              = len(df[sl == "approved"])
    c["Received"]              = len(df[sl == "received"])
    c["In Transit"]            = len(df[df["tracking_status"].str.lower().str.contains("routing delivery|out for delivery", na=False)])
    c["Submitted"]             = len(df[(sl == "approved") & df["tracking_status"].str.lower().str.contains("submitted to courier", na=False)])
    c["Delivered"]             = len(df[(sl == "approved") & df["tracking_status"].str.lower().str.contains("delivered", na=False)])
    c["Courier Cancelled"]     = len(df[df["is_cc"]])
    c["No Tracking"]           = len(df[(sl.isin(["approved", "received"])) & ((df["tracking_number"] == "") | (df["tracking_number"] == "-"))])
    c["Resolution Actioned"]   = len(df[df["resolution_actioned"].str.lower() == "yes"])
    c["No Resolution Actioned"] = len(df[(sl == "received") & (df["resolution_actioned"].str.lower() != "yes")])
    c["Flagged"]               = int(df["is_flagged"].sum())
    c["Issues"]                = c["No Tracking"] + c["Courier Cancelled"] + c.get("Flagged", 0)
    return c

    "Received":            lambda d: d["status"].str.lower() == "received",
    "Courier Cancelled":   lambda d: d["tracking_status"].str.lower().str.contains("courier cancelled", na=False),
    "Approved > Submitted":lambda d: (d["status"].str.lower() == "approved") & d["tracking_status"].str.lower().str.contains("submitted to courier", na=False),
    "Approved > Delivered":lambda d: (d["status"].str.lower() == "approved") & d["tracking_status"].str.lower().str.contains("delivered", na=False),
    "No Tracking":         lambda d: (d["status"].str.lower().isin(["approved","received"])) & ((d["tracking_number"] == "") | (d["tracking_number"] == "-")),
    "Resolution Actioned": lambda d: d["resolution_actioned"].str.lower() == "yes",
    "No Resolution Actioned": lambda d: (d["status"].str.lower() == "received") & (d["resolution_actioned"].str.lower() != "yes"),
    "In Transit":          lambda d: d["tracking_status"].str.lower().str.contains("routing delivery|out for delivery", na=False),
    "Issues":              lambda d: (
        ((d["status"].str.lower().isin(["approved","received"])) & ((d["tracking_number"]=="") | (d["tracking_number"]=="-"))) |
        d["tracking_status"].str.lower().str.contains("courier cancelled", na=False) |
        d["is_flagged"]
    ),
    "Flagged": lambda d: d["is_flagged"],
}


def apply_filters(df: pd.DataFrame) -> pd.DataFrame:
    if df.empty or not st.session_state.get("active_filters"):
        return df
    mask = pd.Series(False, index=df.index)
    for f in st.session_state.active_filters:
        if f in FILTER_MAP:
            mask |= FILTER_MAP[f](df)
    return df[mask]


def apply_additional_filters(df: pd.DataFrame) -> pd.DataFrame:
    if df.empty:
        return df
    d = df.copy()
    if st.session_state.get("status_multi"):
        d = d[d["status"].isin(st.session_state.status_multi)]
    if st.session_state.get("res_multi"):
        d = d[d["resolution_type"].isin(st.session_state.res_multi)]
    actioned = st.session_state.get("actioned_multi", [])
    if set(actioned) == {"Yes"}:
        d = d[d["resolution_actioned"].str.lower() == "yes"]
    elif set(actioned) == {"No"}:
        d = d[d["resolution_actioned"].str.lower() != "yes"]
    if st.session_state.get("tracking_multi"):
        d = d[d["tracking_status"].isin(st.session_state.tracking_multi)]
    q = st.session_state.get("search_query_input", "").strip().lower()
    if q:
        mask = (
            d["rma_id"].astype(str).str.lower().str.contains(q, na=False) |
            d["order_name"].astype(str).str.lower().str.contains(q, na=False) |
            d["tracking_number"].astype(str).str.lower().str.contains(q, na=False) |
            d["customer_name"].astype(str).str.lower().str.contains(q, na=False)
        )
        d = d[mask]
    return d


def toggle_filter(f: str):
    if "active_filters" not in st.session_state:
        st.session_state.active_filters = set()
    if f in st.session_state.active_filters:
        st.session_state.active_filters.discard(f)
    else:
        st.session_state.active_filters.add(f)


def clear_all_filters():
    st.session_state.status_multi = []
    st.session_state.res_multi    = []
    st.session_state.actioned_multi = []
    st.session_state.tracking_multi = []
    st.session_state.search_query_input = ""
    st.session_state.active_filters = set()

@st.dialog("RMA Actions", width="large")
def show_rma_actions_dialog(row_data: pd.Series, store_key: str):
    raw_rma_id = row_data.get("RMA ID", "")
    rma_id = raw_rma_id.split("rmaid=")[-1] if "rmaid=" in str(raw_rma_id) else str(raw_rma_id)
    shipment_id = safe_get(row_data.get("json_data", {}), "shipments.0.shipmentId")
    existing_tracking = row_data.get("tracking_number", "")
    cfg = STORE_CONFIG[store_key]

    st.markdown(f"**{cfg['display_name']} — RMA:** `{rma_id}`")
    tab_comments, tab_tracking, tab_details = st.tabs(["💬 Comment Timeline", "🚚 Update Tracking", "📋 RMA Details"])

    with tab_comments:
        comments = safe_get(row_data.get("json_data", {}), "comments", []) or []
        with st.container(height=350):
            if not comments:
                st.info("No comments for this RMA.")
            else:
                for c in sorted(comments, key=lambda x: x.get("datetime", ""), reverse=True):
                    dt_str = format_date(safe_parse_date_iso(c.get("datetime")))
                    author = c.get("triggeredBy", "System")
                    st.markdown(f"**{dt_str}** · *{author}*")
                    st.markdown(f"> {c.get('htmlText','')}", unsafe_allow_html=True)
                    st.markdown("---")
        with st.form(f"comment_form_{rma_id}"):
            new_comment = st.text_area("Add comment", height=80)
            if st.form_submit_button("Post Comment"):
                if new_comment.strip():
                    ok = post_rma_comment(store_key, rma_id, new_comment)
                    if ok:
                        st.success("Posted!")
                        append_ops_log(f"Comment posted to RMA {rma_id} [{cfg['display_name']}]")
                        st.cache_data.clear()
                        st.rerun()
                    else:
                        st.error("Failed to post comment.")

    with tab_tracking:
        new_tn = st.text_input("New Tracking Number (OPT-)", value=existing_tracking)
        if st.button("Submit Tracking Update"):
            if not shipment_id:
                st.error("No shipment ID found. A shipment must be created in ReturnGO before tracking can be updated. This usually happens when the RMA is approved.")
            elif new_tn and new_tn.strip():
                ok, msg = push_tracking_update(store_key, rma_id, shipment_id, new_tn)
                if ok:
                    st.success(f"Tracking updated to {new_tn}")
                    append_ops_log(f"Tracking updated: RMA {rma_id} → {new_tn} [{cfg['display_name']}]")
                    st.cache_data.clear()
                    st.rerun()
                else:
                    st.error(f"Failed: {msg}")
            else:
                st.warning("Tracking number cannot be empty.")

    with tab_details:
        st.json(row_data.get("json_data", {}))

    """Excel-friendly UTF-8 CSV bytes with BOM for safer delimiter parsing."""
    return df.to_csv(index=False, lineterminator="\n").encode("utf-8-sig")

            f'{rec["status_emoji"]} {rec["status_label"]}</span>')


def render_inbound_card(rec: dict):
    with st.expander(
        f'{rec["status_emoji"]}  {rec["client_ref"]}  ·  {rec["status_label"]}  ·  {rec["item_count"]} SKU  ·  {rec["expected_units"]} units',
        expanded=False,
    ):
        c1, c2, c3, c4 = st.columns(4)
        for col, lbl, val in [
            (c1, "CLIENT REF",       rec["client_ref"]),
            (c2, "SUPPLIER REF",     rec["supplier_ref"]),
            (c3, "EST. ARRIVAL",     rec["est_arrival"]),
            (c4, "LAST EVENT",       rec["last_event_str"]),
        ]:
            with col:
                st.markdown(
                    f'<div style="font-family:var(--mono);font-size:0.62rem;color:var(--muted);text-transform:uppercase;">{lbl}</div>'
                    f'<div style="font-family:var(--mono);font-size:0.85rem;color:var(--text);">{val}</div>',
                    unsafe_allow_html=True,
                )
        if rec["status_code"] == 200:
            st.markdown(
                f'<div style="background:rgba(245,158,11,0.08);border:1px solid rgba(245,158,11,0.3);border-radius:6px;'
                f'padding:0.5rem 0.9rem;font-family:var(--mono);font-size:0.76rem;color:#fbbf24;margin:0.6rem 0;">'
                f'🚚 Awaiting — Supplier Ref: <b>{rec["supplier_ref"]}</b> · Courier: <b>{rec["courier"]}</b> · Waybill: <b>{rec["waybill"]}</b>'
                f'</div>',
                unsafe_allow_html=True,
            )
        if rec["is_variance"]:
            st.markdown(
                f'<div style="background:rgba(239,68,68,0.07);border:1px solid rgba(239,68,68,0.3);border-radius:6px;'
                f'padding:0.5rem 0.9rem;font-family:var(--mono);font-size:0.76rem;color:#f87171;margin:0.6rem 0;">'
                f'⚠️ VARIANCE — Quantities received do not match manifest. Executive review required.'
                f'</div>',
                unsafe_allow_html=True,
            )
        items = rec.get("raw_items", [])
        if items:
            rows_html = ""
            for item in items:
                exp  = item.get("qty", 0) or 0
                rcvd = item.get("receivedQty", 0) or 0
                var  = rcvd != 0 and rcvd != exp
                row_cls = ' class="var-row"' if var else ""
                rcvd_txt = f"<span class=var-flag>⚠️ {rcvd}</span>" if var else (rcvd or "—")
                rows_html += (
                    f'<tr{row_cls}>'
                    f'<td>{item.get("itemNo","—")}</td>'
                    f'<td>{item.get("name","—") or "—"}</td>'
                    f'<td style="text-align:right">{exp}</td>'
                    f'<td style="text-align:right">{rcvd_txt}</td>'
                    f'</tr>'
                )
            st.markdown(
                f'<table class="items-table"><thead><tr>'
                f'<th>SKU</th><th>Name</th><th style="text-align:right">Expected</th><th style="text-align:right">Received</th>'
                f'</tr></thead><tbody>{rows_html}</tbody></table>',
                unsafe_allow_html=True,
            )
        events = rec.get("raw_events", [])
        if events:
            with st.expander("📋 Event Timeline", expanded=False):
                tl = ""
                for ev in events:
                    code  = ev.get("code", "")
                    meta  = PN_INBOUND_STATUS.get(code, {"color": "#6b7280", "emoji": "•"})
                    tl += (
                        f'<div style="padding:3px 0;border-bottom:1px solid rgba(255,255,255,0.04);'
                        f'font-family:var(--mono);font-size:0.7rem;">'
                        f'<span style="color:{meta["color"]}">{meta["emoji"]} {ev.get("description","")}</span>'
                        f'<span style="color:var(--muted);float:right">{_fmt_pn_ts(ev.get("timeStamp",""))}</span>'
                        f'</div>'
                    )
                st.markdown(f'<div>{tl}</div>', unsafe_allow_html=True)
        st.markdown(
            f'<div style="font-family:var(--mono);font-size:0.62rem;color:var(--dim);margin-top:0.6rem;">'
            f'PN ID: {rec["pn_id"]} · Type: {rec["inbound_type"]} · Created: {rec["create_date"]}'
            f'</div>',
            unsafe_allow_html=True,
        )


def render_inbound_tab(store_key: str):
    """Parcelninja Inbound Control Centre — integrated as a tab."""
    cfg = STORE_CONFIG[store_key]

    col_days, col_refresh, col_info = st.columns([1, 1, 4])
    with col_days:
        days = st.selectbox("Window", [7, 14, 30, 60], index=1,
                            format_func=lambda d: f"Last {d} days",
                            key=f"ib_days_{store_key}", label_visibility="collapsed")
    with col_refresh:
        do_fetch = st.button("🔄 Fetch Inbounds", key=f"ib_refresh_{store_key}", type="primary")

    cache_key = f"pn_records_{store_key}"

    if do_fetch or cache_key not in st.session_state:
        with st.spinner(f"Fetching Parcelninja inbounds for {cfg['display_name']}…"):
            st.session_state[cache_key] = fetch_pn_inbounds_enriched(store_key, days)
        append_ops_log(f"Parcelninja inbounds fetched for {cfg['display_name']}: {len(st.session_state[cache_key])} records")

    records: List[dict] = st.session_state.get(cache_key, [])

    if not records:
        st.info("No inbounds found. Click Fetch Inbounds to load data.")
        return

    # KPI row
    total_units  = sum(r["expected_units"] for r in records)
    in_progress  = [r for r in records if r["is_in_progress"]]
    variances    = [r for r in records if r["is_variance"]]
    awaiting     = [r for r in records if r["status_code"] == 200]
    complete     = [r for r in records if r["is_complete"] and not r["is_variance"]]

    kc = st.columns(5)
    for col, cls, lbl, val, sub in [
        (kc[0], "#3b82f6", "Total Inbounds",  str(len(records)),      f"Last {days} days"),
        (kc[1], "#8b5cf6", "Expected Units",  f"{total_units:,}",     "All inbounds"),
        (kc[2], "#f59e0b", "In Progress",     str(len(in_progress)),  "Status < 203"),
        (kc[3], "#ef4444", "Variances",       str(len(variances)),    "Code 204"),
        (kc[4], "#10b981", "Complete",        str(len(complete)),     "Code 203"),
    ]:
        with col:
            st.markdown(
                f'<div style="background:var(--bg3);border:1px solid var(--border2);border-top:2px solid {cls};'
                f'border-radius:7px;padding:0.8rem 1rem;">'
                f'<div style="font-family:var(--mono);font-size:0.6rem;color:var(--muted);text-transform:uppercase;'
                f'letter-spacing:0.09em;">{lbl}</div>'
                f'<div style="font-family:var(--mono);font-size:1.7rem;font-weight:700;color:{cls};">{val}</div>'
                f'<div style="font-family:var(--mono);font-size:0.62rem;color:var(--dim);">{sub}</div>'
                f'</div>',
                unsafe_allow_html=True,
            )

    # Variance alert
    if variances:
        refs = ", ".join(r["client_ref"] for r in variances[:5])
        st.markdown(
            f'<div class="alert-variance">⚠️ <b>{len(variances)} VARIANCE(S) — EXECUTIVE REVIEW REQUIRED</b><br>'
            f'References: {refs}{"…" if len(variances) > 5 else ""}</div>',
            unsafe_allow_html=True,
        )

    # In Progress section
    if in_progress:
        st.markdown('<div class="section-label" style="color:#f59e0b;border-color:rgba(245,158,11,0.2);">⏳ In Progress</div>',
                    unsafe_allow_html=True)
        for r in in_progress:
            render_inbound_card(r)

    # All inbounds
    st.markdown('<div class="section-label">📋 All Inbounds</div>', unsafe_allow_html=True)
    sf_col, sort_col, count_col = st.columns([2, 2, 2])
    with sf_col:
        sf = st.selectbox("Filter", ["All", "Awaiting Arrival", "Arrived at Warehouse",
                                     "Being Processed", "Processing Complete",
                                     "Complete — Variance", "In Progress Only"],
                          key=f"ib_sf_{store_key}", label_visibility="collapsed")
    with sort_col:
        sort = st.selectbox("Sort", ["Most Recent", "Awaiting First", "Variance First", "Units ↓"],
                            key=f"ib_sort_{store_key}", label_visibility="collapsed")

    filtered = records[:]
    if sf == "In Progress Only":
        filtered = [r for r in filtered if r["is_in_progress"]]
    elif sf != "All":
        filtered = [r for r in filtered if r["status_label"] == sf]
    if sort == "Awaiting First":
        filtered.sort(key=lambda r: (0 if r["status_code"] == 200 else 1))
    elif sort == "Variance First":
        filtered.sort(key=lambda r: (0 if r["is_variance"] else 1))
    elif sort == "Units ↓":
        filtered.sort(key=lambda r: r["expected_units"], reverse=True)

    with count_col:
        st.markdown(
            f'<div style="font-family:var(--mono);font-size:0.7rem;color:var(--muted);padding-top:0.45rem;">'
            f'{len(filtered)} of {len(records)}</div>',
            unsafe_allow_html=True,
        )

    for r in filtered:
        render_inbound_card(r)

    # Export
    with st.expander("📊 Export as Table"):
        df_exp = pd.DataFrame([{
            "Client Ref": r["client_ref"], "Supplier Ref": r["supplier_ref"],
            "Status": r["status_label"], "Expected Units": r["expected_units"],
            "SKU Count": r["item_count"], "Est. Arrival": r["est_arrival"],
            "Waybill": r["waybill"], "Courier": r["courier"],
            "Last Event": r["last_event_str"],
        } for r in filtered])
        st.dataframe(df_exp, width='stretch', hide_index=True)
        st.download_button("⬇️ Export CSV", dataframe_to_csv_bytes(df_exp),
                           f"{store_key}_inbounds_{datetime.now().strftime('%Y%m%d_%H%M')}.csv", "text/csv",
                           key=f"ib_csv_{store_key}")



def render_consolidated_tab():
    st.markdown('<div class="section-label">🏢 Consolidated RMA Primary Statuses (All Stores)</div>', unsafe_allow_html=True)

    c1, c2, c_spacer = st.columns([2, 2, 6])
    with c1:
        if st.button("🔄 Load / Refresh All Stores", key="refresh_all_stores", type="primary", use_container_width=True):
            with st.spinner("Refreshing all stores for consolidated view…"):
                for sk in STORE_CONFIG:
                    st.cache_data.clear()
                    st.session_state[f"rma_df_{sk}"] = fetch_and_cache_store_data(sk)
                    st.session_state[f"rma_df_enriched_{sk}"] = None
            st.session_state["last_sync_time"] = datetime.now(timezone.utc)
            append_ops_log("Consolidated refresh completed for all stores")
            st.rerun()
    with c2:
        if st.button("⚙️ Sync All Updates", key="sync_all_updates", use_container_width=True):
            with st.spinner("Incrementally syncing all stores..."):
                for sk in STORE_CONFIG:
                    perform_incremental_sync(sk)
            
            fetch_and_cache_store_data.clear()
            append_ops_log("Incremental sync for all stores completed. Main cache cleared for next reload.")
            st.rerun()

    # Ensure raw data is loaded for all stores sequentially first, as fetch_and_cache_store_data uses st.progress
    for sk, scfg in STORE_CONFIG.items():
        if f"rma_df_{sk}" not in st.session_state or st.session_state[f"rma_df_{sk}"] is None:
            st.session_state[f"rma_df_{sk}"] = fetch_and_cache_store_data(sk)
            st.session_state[f"rma_df_enriched_{sk}"] = None
            append_ops_log(f"Loaded {scfg['display_name']} data for consolidated view")

    all_frames: List[pd.DataFrame] = []
    
    def enrich_store_data(store_key_config_tuple):
        """Fetches enriched data for a store; designed for use with ThreadPoolExecutor."""
        store_key, store_config = store_key_config_tuple
        enriched_df = get_enriched_store_df(store_key)
        if enriched_df is not None and not enriched_df.empty:
            df = enriched_df.copy()
            df["Store"] = store_config["display_name"]
            return df
        return None

    total_stores = len(STORE_CONFIG)
    with st.status(f"Analyzing data for {total_stores} stores...", expanded=True) as status:
        for i, store_key_config_tuple in enumerate(STORE_CONFIG.items()):
            store_key, store_config = store_key_config_tuple
            status.update(label=f"({i+1}/{total_stores}) Analyzing data for {store_config['display_name']}...")
            df = enrich_store_data(store_key_config_tuple)
            if df is not None:
                all_frames.append(df)
        status.update(label="All stores analyzed!", state="complete")

    all_frames = [df for df in all_frames if not df.empty]
    if not all_frames:
        st.info("No consolidated data loaded yet. Click **Load / Refresh All Stores**.")
        return

    df_all = pd.concat(all_frames, ignore_index=True)

    # Re-introduce summary table with timestamps
    st.markdown('<div class="section-label">🏢 Store Sync Status</div>', unsafe_allow_html=True)
    status_order = ["Pending", "Approved", "Received"]
    summary = (
        df_all[df_all["status"].isin(status_order)]
        .groupby(["Store", "status"])
        .size()
        .unstack(fill_value=0)
        .reindex(columns=status_order, fill_value=0)
        .reset_index()
    )
    summary["Total Open"] = summary[status_order].sum(axis=1)

    timestamps = load_sync_timestamps()
    store_name_to_key = {v['display_name']: k for k, v in STORE_CONFIG.items()}
    
    def get_formatted_timestamp(store_name: str) -> str:
        store_key = store_name_to_key.get(store_name)
        if store_key:
            ts = timestamps.get(store_key)
            if ts:
                return ts.strftime("%Y-%m-%d %H:%M")
        return "Never"

    summary["Last Synced"] = summary["Store"].apply(get_formatted_timestamp)
    
    # Reorder columns to have Last Synced after Store
    cols = summary.columns.tolist()
    summary = summary[[cols[0], cols[-1]] + cols[1:-1]]

    st.dataframe(summary, width='stretch', hide_index=True)

    status_views = [
        ("Pending", lambda d: d["status"].str.lower() == "pending"),
        ("Approved", lambda d: d["status"].str.lower() == "approved"),
        ("Received", lambda d: d["status"].str.lower() == "received"),
        ("No Tracking", lambda d: (
            d["status"].str.lower().isin(["approved", "received"]) &
            ((d["tracking_number"] == "") | (d["tracking_number"] == "-"))
        )),
        ("Courier Cancelled", lambda d: d["tracking_status"].str.lower().str.contains("courier cancelled", na=False)),
        ("Submitted to Courier", lambda d: (
            (d["status"].str.lower() == "approved") &
            d["tracking_status"].str.lower().str.contains("submitted to courier", na=False)
        )),
        ("In Transit", lambda d: d["tracking_status"].str.lower().str.contains("routing delivery|out for delivery", na=False)),
        ("Delivered", lambda d: (
            (d["status"].str.lower() == "approved") &
            d["tracking_status"].str.lower().str.contains("delivered", na=False)
        )),
    ]

    st.markdown('<div class="section-label">📦 Consolidated Status Blocks</div>', unsafe_allow_html=True)
    if "consolidated_table_scope" not in st.session_state:
        st.session_state["consolidated_table_scope"] = {"status": None, "store": None}

    for row_start in range(0, len(status_views), 4):
        row_statuses = status_views[row_start:row_start + 4]
        row_cols = st.columns(4)
        for idx, (status_name, status_mask) in enumerate(row_statuses):
            scope = df_all[status_mask(df_all)].copy()
            total_count = len(scope)
            by_store = (
                scope.groupby("Store").size().reindex(
                    [cfg["display_name"] for cfg in STORE_CONFIG.values()],
                    fill_value=0,
                )
            )
            with row_cols[idx]:
                st.metric(status_name, f"{total_count:,}")
                with st.expander(f"Per store breakdown — {status_name}", expanded=True):
                    breakdown_df = by_store.reset_index()
                    breakdown_df.columns = ["Store", "RMA Count"]

                    # Add 'All' row
                    all_row = pd.DataFrame([{"Store": "All", "RMA Count": total_count}])
                    breakdown_df = pd.concat([all_row, breakdown_df], ignore_index=True)
                    
                    clickable_df = breakdown_df[breakdown_df["RMA Count"] > 0].reset_index(drop=True)

                    if not clickable_df.empty:
                        selection = st.dataframe(
                            clickable_df,
                            width='stretch',
                            hide_index=True,
                            on_select="rerun",
                            selection_mode="single-row",
                            key=f"breakdown_table_{status_name}"
                        )

                        if isinstance(selection, dict):
                            sel_content = selection.get("selection", {})
                            sel_rows = sel_content.get("rows")

                            if sel_rows:
                                selected_row_index = sel_rows[0]
                                selected_store = clickable_df.iloc[selected_row_index]["Store"]

                                # Toggle filter
                                current_scope = st.session_state["consolidated_table_scope"]

                                new_scope_store = None
                                if selected_store != "All":
                                    new_scope_store = selected_store

                                if current_scope.get("status") == status_name and current_scope.get("store") == new_scope_store:
                                    st.session_state["consolidated_table_scope"] = {"status": None, "store": None}
                                else:
                                    st.session_state["consolidated_table_scope"] = {
                                        "status": status_name,
                                        "store": new_scope_store
                                    }
                                st.rerun()
                    else:
                        # Display non-clickable version if no items with that status
                        st.dataframe(breakdown_df, width='stretch', hide_index=True)

    st.markdown('<div class="section-label">📋 Consolidated Data Table</div>', unsafe_allow_html=True)
    disp = df_all.copy()

    active_scope = st.session_state.get("consolidated_table_scope", {"status": None, "store": None})
    scoped_status = active_scope.get("status")
    scoped_store = active_scope.get("store")
    if scoped_status:
        status_view_map = {name: mask for name, mask in status_views}
        mask_fn = status_view_map.get(scoped_status)
        if mask_fn is not None:
            mask = mask_fn(disp)
            if scoped_store:
                mask = mask & (disp["Store"] == scoped_store)
            disp = disp[mask].copy()

            top_col, _, _ = st.columns([4, 1, 1])
            with top_col:
                filter_text = f"Filtered view: **{scoped_status}**"
                if scoped_store:
                    filter_text += f" · **{scoped_store}**"
                else:
                    filter_text += " · **All Stores**"
                st.info(filter_text)
            if st.button("Clear consolidated filter", key="clear_consolidated_scope"):
                st.session_state["consolidated_table_scope"] = {"status": None, "store": None}
                st.rerun()

    disp.insert(0, "No", range(1, len(disp) + 1))
    disp = disp.rename(columns={
        "rma_id": "RMA ID", "order_name": "Order", "customer_name": "Customer Name",
        "client_id": "Client ID", "inbound_status": "Inbound Status", "inbound_update_date": "Inbound update Date", "status": "Current Status",
        "tracking_number": "Tracking Number", "tracking_status": "Tracking Status",
        "requested_date": "Requested date", "approved_date": "Approved date", "received_date": "Received date",
        "days_since_requested": "Days since requested", "days_since_credit": "Days since credit",
        "last_rma_update": "Last RMA Update", "resolution_type": "resolutionType",
        "resolution_actioned": "Resolution actioned",
    })
    for col in ("Requested date", "Approved date", "Received date"):
        if col in disp.columns:
            disp[col] = disp[col].apply(lambda x: format_date(x) if pd.notna(x) else "-")

    display_cols = [
        "No", "Store", "RMA ID", "Order", "Customer Name", "Client ID", "Inbound Status", "Inbound update Date",
        "Current Status", "Tracking Number", "Tracking Status", "Requested date", "Approved date", "Received date",
        "Days since requested", "Days since credit", "Last RMA Update", "resolutionType", "Resolution actioned", "failures",
    ]
    for c in display_cols:
        if c not in disp.columns:
            disp[c] = "-"
    disp["RMA ID"] = disp["RMA ID"].apply(
        lambda x: f"https://app.returngo.ai/dashboard/returns?filter_status=open&rmaid={x}" if x else "-"
    )
    disp["Tracking Number"] = disp["Tracking Number"].apply(lambda x: str(x) if x not in (None, "") else "-")

    selection = st.dataframe(
        disp[display_cols],
        width='stretch',
        hide_index=True,
        height=600,
        on_select="rerun",
        selection_mode="single-row",
        key="consolidated_table"
    )

    # Action panel for selected row
    sel_dict = selection.get("selection", {})
    sel_rows = sel_dict.get("rows", [])

    if sel_rows:
        selected_row_index = sel_rows[0]
        selected_rma_row = disp.iloc[selected_row_index]
        rma_id = str(selected_rma_row["RMA ID"])
        store_name = selected_rma_row["Store"]

        store_name_to_key = {v['display_name']: k for k, v in STORE_CONFIG.items()}
        store_key = store_name_to_key.get(store_name)

        if store_key:
            st.markdown("---")
            col1, col2 = st.columns([3, 1])
            with col1:
                st.markdown(f"**Selected:** RMA `{rma_id}` from **{store_name}**")
            with col2:
                if st.button(f"🔄 Refresh this RMA", key=f"refresh_rma_{rma_id}", use_container_width=True):
                    with st.spinner(f"Refreshing RMA {rma_id}..."):
                        success = update_single_rma(rma_id, store_key)
                    if success:
                        st.success(f"RMA {rma_id} has been updated.")
                        time.sleep(1)
                        st.rerun()
                    else:
                        st.error(f"Could not update RMA {rma_id}.")
            st.markdown("---")

    st.download_button(
        "⬇️ Download consolidated CSV",
        dataframe_to_csv_bytes(disp[display_cols]),
        f"bounty_consolidated_rmas_{datetime.now().strftime('%Y%m%d_%H%M')}.csv",
        "text/csv",
        key="dl_consolidated",
    )

            '<div style="position:fixed;top:0;left:0;width:100%;height:100%;background:rgba(0,0,0,0.95);'
            'z-index:9998;display:flex;align-items:center;justify-content:center;text-align:center;">'
            '<div><h2 style="color:#ef4444">Session Disconnected</h2>'
            '<p style="color:#94a3b8">All processing stopped. Close the browser tab.</p></div></div>',
            unsafe_allow_html=True,
        )
        st.stop()

    store_key = st.session_state["active_store"]
    cfg = STORE_CONFIG.get(store_key, STORE_CONFIG["diesel"]) # Default to diesel if key is bad
    inject_css(cfg["accent"], cfg["accent_dark"])

    # ── SIDEBAR ──────────────────────────────────
    with st.sidebar:
        st.markdown("### 🏢 Bounty Brands")

        is_consolidated = st.session_state.get("active_workspace", "consolidated") == "consolidated"
        if st.button("🏢 Consolidated View", key="nav_consolidated", width='stretch', type="primary" if is_consolidated else "secondary"):
            if st.session_state.get("active_workspace") != "consolidated":
                st.session_state["active_workspace"] = "consolidated"
                st.session_state["active_filters"] = set()
                st.rerun()

        st.markdown("---")

        for sk, scfg in STORE_CONFIG.items():
            is_active = sk == store_key and st.session_state.get("active_workspace") == "store"
            btn_label = f'{scfg["icon"]} {scfg["display_name"]}'
            btn_type  = "primary" if is_active else "secondary"
            if st.button(btn_label, key=f"nav_{sk}", width='stretch', type=btn_type):
                if sk != st.session_state.get("active_store") or st.session_state.get("active_workspace") != "store":
                    st.session_state["active_store"] = sk
                    st.session_state["active_workspace"] = "store"
                    st.session_state["active_filters"] = set()
                    st.rerun()
        
        st.markdown("---")

        # Sync controls
        st.markdown("**Cache & Sync**")
        active_workspace = st.session_state.get("active_workspace", "store")

        if active_workspace == 'store':
            if st.button("🔄 Full Refresh", width='stretch', help="Clear cache and re-fetch all data for this store."):
                append_ops_log(f"Starting full refresh for {cfg['display_name']}...")
                st.cache_data.clear()
                st.session_state[f"rma_df_{store_key}"] = None
                st.session_state[f"rma_df_enriched_{store_key}"] = None
                append_ops_log(f"Cache cleared for {cfg['display_name']}.")
                st.rerun()
            
            if st.button("⚙️ Sync Updates", width='stretch', help="Fetch only data updated since the last sync for this store."):
                append_ops_log(f"Starting incremental sync for {cfg['display_name']}...")
                with st.spinner(f"Fetching updates for {cfg['display_name']}..."):
                    perform_incremental_sync(store_key)
                st.rerun()

            if st.button("🚚 Refresh Courier Status", width='stretch'):
                df_current = st.session_state.get(f"rma_df_{store_key}")
                if df_current is not None and not df_current.empty:
                    enriched = enrich_rma_dataframe(df_current, store_key)
                    updated  = force_courier_refresh(enriched)
                    st.session_state[f"rma_df_{store_key}"] = updated
                    st.session_state[f"rma_df_enriched_{store_key}"] = updated
                    append_ops_log(f"Courier status refreshed for {cfg['display_name']}.")
                    st.rerun()
                else:
                    st.warning("No data loaded. Do a full refresh first.")

        st.markdown("---")
        st.markdown("##### Sync Integrity Check")

        if active_workspace == "store":
            store_name = STORE_CONFIG[store_key]['display_name']

            sync_check_status = st.selectbox(
                "Status to Check",
                ["Pending", "Approved", "Received"],
                key="sync_check_status",
                help="Select a status to check for discrepancies between the dashboard and ReturnGO."
            )

            if st.button(f"Check '{sync_check_status}' for {store_name}", key="run_sync_check"):
                df_key = f"rma_df_{store_key}"
                is_data_loaded = df_key in st.session_state and st.session_state.get(df_key) is not None

                with st.spinner(f"Checking {sync_check_status} count for {store_name}..."):
                    api_rmas = fetch_returngo_rmas(store_key, status=sync_check_status)
                    api_count = len(api_rmas)
                    local_count = 0
                    if is_data_loaded:
                        store_df = get_enriched_store_df(store_key)
                        if not store_df.empty:
                            local_count = len(store_df[store_df["status"].str.lower() == sync_check_status.lower()])
                
                st.info(f"Count for '{sync_check_status}':\n- ReturnGO API: **{api_count}**\n- Dashboard: **{local_count}**")

                if api_count != local_count:
                    st.warning("Discrepancy detected. Resolving...")
                    if not is_data_loaded:
                        append_ops_log(f"Discrepancy check on {store_name}: No local data found. Performing full sync.")
                        with st.spinner(f"Performing FULL sync for {store_name}..."):
                            fetch_and_cache_store_data.clear()
                            st.session_state[df_key] = fetch_and_cache_store_data(store_key)
                            st.session_state[f"rma_df_enriched_{store_key}"] = None
                    else:
                        append_ops_log(f"Discrepancy check on {store_name}: Mismatch found. Performing incremental sync.")
                        with st.spinner(f"Performing incremental sync for {store_name}..."):
                            perform_incremental_sync(store_key)
                    
                    append_ops_log("Sync complete. Rerunning.")
                    st.success("Sync complete. Data should now be consistent.")
                    time.sleep(2)
                    st.rerun()
                else:
                    st.success("Counts match. Data is in sync.")
        else:
            st.info("Select an individual store to run this check.")

        st.markdown("---")
        st.markdown("**📋 System & Performance**")
        if st.button("Show Logs & Metrics", key="show_log_dialog", width='stretch'):
            show_data_table_log()
        
        st.markdown("---")
        if st.button("⏻ Shutdown", type="primary", width='stretch'):
            st.session_state.disconnected = True
            st.rerun()

    # ... (rest of main function)
    active_workspace = st.session_state.get("active_workspace", "store")

    # ── MASTHEAD ─────────────────────────────────
    now_str = datetime.now().strftime("%d %b %Y  %H:%M")
    masthead_title = "🏢 Consolidated View — All Five Brands" if active_workspace == "consolidated" else f'{cfg["icon"]} {cfg["display_name"]}'
    st.markdown(
        f'<div class="masthead">'
        f'<div><div class="masthead-brand">{masthead_title}</div>'
        f'<div class="masthead-sub">Bounty Brands · ReturnGO Ops Dashboard · {now_str} SAST</div></div>'
        f'</div>',
        unsafe_allow_html=True,
    )

    connection_text = (
        "🟢 Connected · Consolidated scope across all stores"
        if active_workspace == "consolidated"
        else f'🟢 Connected · {cfg["shopify_url"]} · Parcelninja Store: {cfg["pn_store_id"][:8]}…'
    )
    st.markdown(f'<div class="connection-status">{connection_text}</div>', unsafe_allow_html=True)

    if active_workspace == "consolidated":
        render_consolidated_tab()
        return

    # ── LOAD DATA ────────────────────────────────
    df_key = f"rma_df_{store_key}"
    if st.session_state.get(df_key) is None:
        st.session_state[df_key] = fetch_and_cache_store_data(store_key)
        st.session_state[f"rma_df_enriched_{store_key}"] = None
        st.session_state["last_sync_time"] = datetime.now(timezone.utc)
        append_ops_log(f"Data loaded for {cfg['display_name']}: {len(st.session_state[df_key])} records")

    df_raw = st.session_state[df_key]
    with st.spinner(f"Analyzing data for {cfg['display_name']}... This may take a moment."):
        df_enriched = get_enriched_store_df(store_key)
        counts = compute_counts(df_enriched)

    # ── MAIN TABS ─────────────────────────────────
    tab_rma, tab_inbound = st.tabs(["📊 RMA Command Centre", "📦 Inbound Control Centre"])

    with tab_rma:
        st.markdown('<div class="section-label">📅 Requested Timeline</div>', unsafe_allow_html=True)
        if not df_enriched.empty and "requested_date" in df_enriched.columns:
            tl_df = df_enriched[df_enriched["status"].isin(ACTIVE_RG_STATUSES)].copy()
            tl_df["_dt"] = pd.to_datetime(tl_df["requested_date"], errors="coerce")
            tl_df = tl_df.dropna(subset=["_dt"])
            if not tl_df.empty:
                chart_df = tl_df.groupby(tl_df["_dt"].dt.date).size().reset_index(name="RMAs")
                chart_df.columns = ["Date", "RMAs"]
                st.line_chart(chart_df.set_index("Date"), height=220, color=cfg["accent"])
        else:
            st.info("No timeline data available.")

        st.markdown('<div class="section-label">📊 RMA Status Overview</div>', unsafe_allow_html=True)
        # Metric cards rendering...
        row1 = st.columns(4)
        render_metric_card(row1[0], "Total Open",      "Total Open",   "All",               counts)
        render_metric_card(row1[1], "Pending",          "Pending",      "Pending Requests",  counts)
        render_metric_card(row1[2], "In Transit",       "In Transit",   "In Transit",        counts)
        render_metric_card(row1[3], "Received",         "Received",     "Received",          counts)
        row2 = st.columns(4)
        render_metric_card(row2[0], "Issues",           "Issues",             "Issues",                   counts)
        render_metric_card(row2[1], "Flagged",          "Flagged",            "Flagged",                  counts)
        render_metric_card(row2[2], "Approved→Submitted","Submitted",         "Approved > Submitted",     counts)
        render_metric_card(row2[3], "Approved→Delivered","Delivered",         "Approved > Delivered",     counts)
        row3 = st.columns(4)
        render_metric_card(row3[0], "Courier Cancelled","Courier Cancelled",  "Courier Cancelled",        counts)
        render_metric_card(row3[1], "No Tracking",      "No Tracking",        "No Tracking",              counts)
        render_metric_card(row3[2], "Resolution Done",  "Resolution Actioned","Resolution Actioned",      counts)
        render_metric_card(row3[3], "No Resolution",    "No Resolution Actioned","No Resolution Actioned",counts)

        st.markdown("---")
        # Search and filter UI...
        sc1, sc2, sc3 = st.columns([6, 2, 2])
        # ...
        
        df_filtered = apply_filters(df_enriched)
        df_filtered = apply_additional_filters(df_filtered)
        
        # Data table rendering...
        if not df_filtered.empty:
            display_df = df_filtered.copy()
            display_df.insert(0, "No", range(1, len(display_df) + 1))
            display_df = display_df.rename(columns={
                "rma_id": "RMA ID",
                "order_name": "Order",
                "customer_name": "Customer Name",
                "client_id": "Client ID",
                "inbound_status": "Inbound Status",
                "inbound_update_date": "Inbound update Date",
                "status": "Current Status",
                "tracking_number": "Tracking Number",
                "tracking_status": "Tracking Status",
                "requested_date": "Requested date",
                "approved_date": "Approved date",
                "received_date": "Received date",
                "days_since_requested": "Days since requested",
                "days_since_credit": "Days since credit",
                "last_rma_update": "Last RMA Update",
                "resolution_type": "resolutionType",
                "is_tracking_updated": "Is Tracking Updated",
                "resolution_actioned": "Resolution actioned",
            })

            for col in ("Requested date", "Approved date", "Received date"):
                if col in display_df.columns:
                    display_df[col] = display_df[col].apply(lambda x: format_date(x) if pd.notna(x) else "-")

            display_cols = [
                "No", "RMA ID", "Order", "Customer Name", "Client ID", "Inbound Status", "Inbound update Date",
                "Current Status", "Tracking Number", "Tracking Status", "Requested date", "Approved date", "Received date",
                "Days since requested", "Days since credit", "Last RMA Update", "resolutionType", "Resolution actioned",
                "Is Tracking Updated", "failures",
            ]

            for c in display_cols:
                if c not in display_df.columns:

            render_data_table(display_df, display_cols, store_key)
        else:

    with tab_inbound:
        render_inbound_tab(store_key)

    last = st.session_state.get("last_sync_time")
    last_str = last.strftime("%Y-%m-%d %H:%M:%S UTC") if last else "Never"
    st.markdown(

        f'<span>🏢 Bounty Brands · ReturnGO + Parcelninja · {cfg["shopify_url"]}</span>'
